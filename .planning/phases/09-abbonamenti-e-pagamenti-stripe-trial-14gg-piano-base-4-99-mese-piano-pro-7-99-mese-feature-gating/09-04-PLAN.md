Create `src/app/abbonamento/page.tsx`. This is a hybrid page: server component wrapper that reads subscription status, with a client component for interactive buttons.

The page structure:
1. Server component reads the user's current subscription via `getSubscription()`
2. Passes subscription status to a client component `PricingCards`
3. PricingCards renders three plan cards in Italian

```typescript
// src/app/abbonamento/page.tsx
import { cookies } from "next/headers";
import { createClient } from "@supabase/supabase-js";
import { getSubscription } from "@/lib/stripe";
import { PricingCards } from "./PricingCards";

export const metadata = {
  title: "Abbonamento | Menumix",
  description: "Scegli il piano giusto per te",
};

export default async function AbbonamentoPage() {
  // Try to get user from Supabase auth cookie
  let userId: string | null = null;
  let userEmail: string | null = null;

  try {
    const cookieStore = await cookies();
    // Read the Supabase auth token from cookies to get user identity
    // We use the service role client to verify the user from the cookie
    const supabase = createClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_ROLE_KEY!,
    );
    // The auth token is in sb-<ref>-auth-token cookie
    // For simplicity, we'll pass null and let PricingCards handle unauthenticated state
  } catch {
    // Not authenticated — show pricing cards without checkout capability
  }

  return (
    <main style={{ maxWidth: 900, margin: "0 auto", padding: "24px 16px" }}>
      <h1 style={{ textAlign: "center", fontSize: 28, marginBottom: 8, color: "var(--olive-dark, #3d4f2f)" }}>
        Scegli il tuo piano
      </h1>
      <p style={{ textAlign: "center", color: "var(--sepia-light, #8b7d6b)", marginBottom: 32, fontSize: 15 }}>
        Inizia con 14 giorni gratuiti, poi scegli il piano che fa per te.
      </p>
      <PricingCards />
    </main>
  );
}
```

    cta: "Abbonati al Piano Pro",
    highlight: true,

export function PricingCards() {
  const { user } = useAuth();
  const [loading, setLoading] = useState<string | null>(null);
  const [error, setError] = useState<string | null>(null);

  const handleSubscribe = async (planId: string) => {
    if (!user) {
      setError("Devi accedere per abbonarti. Vai al profilo per fare login.");
      return;
    }

    setLoading(planId);
    setError(null);
    try {
      const priceId = planId === "base"
        ? process.env.NEXT_PUBLIC_STRIPE_PRICE_ID_BASE!
        : process.env.NEXT_PUBLIC_STRIPE_PRICE_ID_PRO!;
      // Note: priceId env vars must be NEXT_PUBLIC_ for client access, OR
      // pass planId to a server action that reads server-side env vars.
      // Better pattern: pass planType to server action, let it resolve priceId.
      await createCheckoutSession(user.id, user.email!, planId === "base" ? "base" : "pro");
    } catch (err) {
      setError("Errore durante la creazione della sessione di pagamento.");
      setLoading(null);
    }
  };

  return (
    <>
      {error && (
        <p style={{ color: "var(--terra, #c0392b)", textAlign: "center", marginBottom: 16, fontSize: 14 }}>
          {error}
        </p>
      )}
      <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fit, minmax(260px, 1fr))", gap: 20 }}>
        {PLANS.map((plan) => (
          <div
            key={plan.id}
            style={{
              border: plan.highlight ? "2px solid var(--olive, #6b7c45)" : "1px solid var(--sepia-border, #d4c5a9)",
              borderRadius: 12,
              padding: 24,
              background: plan.highlight ? "var(--cream-light, #faf8f0)" : "white",
              position: "relative",
            }}
          >
            {plan.highlight && (
              <span style={{
                position: "absolute", top: -12, left: "50%", transform: "translateX(-50%)",
                background: "var(--olive, #6b7c45)", color: "white", padding: "2px 12px",
                borderRadius: 12, fontSize: 12, fontWeight: 600,
              }}>
                Consigliato
              </span>
            )}
            <h3 style={{ fontSize: 20, marginBottom: 4, color: "var(--olive-dark, #3d4f2f)" }}>{plan.name}</h3>
            <p style={{ fontSize: 28, fontWeight: 700, margin: "8px 0" }}>
              {plan.price}
              <span style={{ fontSize: 14, fontWeight: 400, color: "var(--sepia-light, #8b7d6b)" }}>{plan.period}</span>
            </p>
            <ul style={{ listStyle: "none", padding: 0, margin: "16px 0" }}>
              {plan.features.map((f, i) => (
                <li key={i} style={{ fontSize: 14, padding: "4px 0", color: "var(--sepia, #5c4f3d)" }}>
                  {"\u2713"} {f}
                </li>
              ))}
            </ul>
            {plan.id === "free" ? (
              <p style={{ fontSize: 13, color: "var(--sepia-light, #8b7d6b)", textAlign: "center", marginTop: 16 }}>
                Attivo per tutti i nuovi utenti
              </p>
            ) : (
              <button
                className="btn-primary"
                style={{ width: "100%", marginTop: 8 }}
                disabled={loading === plan.id}
                onClick={() => handleSubscribe(plan.id)}
              >
                {loading === plan.id ? "Caricamento..." : plan.cta}
              </button>
            )}
          </div>
        ))}
      </div>
    </>
  );
}
